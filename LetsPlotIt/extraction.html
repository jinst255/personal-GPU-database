Data Extraction

import polars as pl
import numpy as np
from lets_plot import *
import polars.selectors as cs

LetsPlot.setup_html(isolated_frame=True)
df = pl.read_csv(
    'gpu_database.csv',
    null_values=['null', 'NULL', 'None', ''],
    infer_schema_length=10000,
    schema_overrides={'cuda_cores': pl.Int64}
)
# Filter for used eBay prices, exclude mobile category
df_plot = df.filter(
    (pl.col('price_usd_ebay_used').is_not_null()) & 
    (pl.col('category') != 'mobile')
).select([
    'gpu_name', 'vram_gb', 'price_usd_ebay_used', 'category'
])

# Create scatter plot: VRAM vs Used Price, colored by category
plot = (
    ggplot(df_plot, aes(x='vram_gb', y='price_usd_ebay_used', color='category')) +
    geom_point(size=2.5, alpha=0.7) +
    geom_text(aes(label='gpu_name'), size=1.8, nudge_y=50, nudge_x=0.1, angle=45, alpha=0.7) +
    scale_x_continuous() +
    scale_y_continuous() +
    scale_color_manual(values={'consumer': '#1f77b4', 'professional': '#ff7f0e'}) +
    labs(
        title='GPU Used Price vs VRAM',
        x='VRAM (GB)',
        y='Used Price - eBay (USD)',
        color='Category'
    ) +
    theme_minimal()
)
